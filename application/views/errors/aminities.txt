		// ==================== AMENITIES CHART ====================
		function initAmenitiesChart() {
			const canvas = document.getElementById('amenitiesChart');
			if (!canvas) return;
			const ctx = canvas.getContext('2d');
			if (window.amenitiesChart instanceof Chart) window.amenitiesChart.destroy();

			// Get booking data for last 7 days
			const days = [];
			const bookingCounts = [];

			for (let i = 6; i >= 0; i--) {
				const date = new Date();
				date.setDate(date.getDate() - i);
				const dayName = date.toLocaleString('default', { weekday: 'short' });
				days.push(dayName);

				const dateStr = date.toISOString().split('T')[0];

				const count = bookingData.filter(b => b.bookingDate === dateStr).length;
				bookingCounts.push(count);
			}

			const isMobile = window.innerWidth < 480;
			const isTablet = window.innerWidth >= 480 && window.innerWidth < 768;

			try {
				window.amenitiesChart = new Chart(ctx, {
					type: 'line',
					data: {
						labels: days,
						datasets: [{
							label: 'Bookings',
							data: bookingCounts,
							borderColor: '#3498db',
							backgroundColor: 'rgba(52, 152, 219, 0.1)',
							borderWidth: 3,
							pointRadius: isMobile ? 3 : 4,
							pointHoverRadius: 6,
							tension: 0.3,
							fill: true
						}]
					},
					options: {
						responsive: true,
						maintainAspectRatio: false,
						plugins: {
							legend: { display: false },
							tooltip: {
								callbacks: {
									label: function (context) {
										return `Bookings: ${context.parsed.y}`;
									}
								}
							}
						},
						scales: {
							y: {
								beginAtZero: true,
								ticks: {
									stepSize: 1,
									font: { size: isMobile ? 10 : 11 }
								},
								grid: {
									color: document.body.classList.contains('dark-mode') ? 'rgba(255,255,255,0.1)' : 'rgba(0,0,0,0.05)'
								}
							},

								grid: { display: false },
								ticks: {
									font: { size: isMobile ? 10 : 11 }
								}
							}
						}
					}
				});

		function toggleView(view) {
			currentView = view;
			const gridView = document.getElementById('amenitiesGridView');
			const tableView = document.getElementById('amenitiesTableView');
			const gridBtn = document.getElementById('gridViewBtn');
			const tableBtn = document.getElementById('tableViewBtn');

			if (view === 'grid') {
				gridView.style.display = 'grid';
				tableView.style.display = 'none';
				gridBtn.classList.add('active');
				tableBtn.classList.remove('active');
			} else {
				gridView.style.display = 'none';
				tableView.style.display = 'block';

				gridBtn.classList.remove('active');
			}

		// ==================== FILTER FUNCTION ====================
		function filterAmenities() {
			const search = document.getElementById('amenitySearch')?.value.toLowerCase() || '';
			const statusFilter = document.getElementById('statusFilter')?.value || '';
			const categoryFilter = document.getElementById('categoryFilter')?.value || '';
			const priceFilter = document.getElementById('priceFilter')?.value || '';

			const filtered = amenityData.filter(a => {
				// Search filter
				const matchesSearch = search === '' ||
					a.name.toLowerCase().includes(search) ||
					a.description.toLowerCase().includes(search) ||
					a.location?.toLowerCase().includes(search) ||
					a.amenityId.toLowerCase().includes(search);

				// Status filter
				const matchesStatus = statusFilter === '' || a.status === statusFilter;

				// Category filter
				const matchesCategory = categoryFilter === '' || a.category === categoryFilter;

				// Price filter
				let matchesPrice = true;
				if (priceFilter === 'free') matchesPrice = a.price === 0;
				else if (priceFilter === 'under500') matchesPrice = a.price > 0 && a.price < 500;
				else if (priceFilter === '500-1000') matchesPrice = a.price >= 500 && a.price <= 1000;
				else if (priceFilter === 'above1000') matchesPrice = a.price > 1000;

				return matchesSearch && matchesStatus && matchesCategory && matchesPrice;
			});

			loadAmenitiesGrid(filtered);
			loadAmenitiesTable(filtered);
			updateStats(filtered);
		}

		function confirmBooking() {
			const amenityId = parseInt(document.getElementById('bookingAmenityId').value);
			const residentId = document.getElementById('bookingResidentId').value;
			const bookingDate = document.getElementById('bookingDate').value;
			const startTime = document.getElementById('bookingStartTime').value;
			const duration = parseInt(document.getElementById('bookingDuration').value);
			const guests = parseInt(document.getElementById('bookingGuests').value);
			const requests = document.getElementById('bookingRequests').value;

			if (!amenityId || !residentId || !bookingDate || !startTime || !duration) {
				showNotification('Please fill all required fields', 'error');
				return;
			}

			const amenity = amenityData.find(a => a.id === amenityId);
			if (!amenity) return;

			// Calculate end time
			const [hours, minutes] = startTime.split(':');
			const endHour = parseInt(hours) + duration;
			const endTime = `${endHour.toString().padStart(2, '0')}:${minutes}`;

			// Calculate amount
			const amount = amenity.price * duration;

			const residentMap = {
				'1': { name: 'Rajesh Kumar', flat: 'A-101' },
				'2': { name: 'Priya Sharma', flat: 'A-102' },
				'3': { name: 'Amit Patel', flat: 'B-201' },
				'4': { name: 'Sunita Reddy', flat: 'C-305' },
				'5': { name: 'Vikram Singh', flat: 'D-410' }
			};

			const newId = bookingData.length > 0 ? Math.max(...bookingData.map(b => b.id), 0) + 1 : 1;
			const year = new Date().getFullYear();
			const count = (bookingData.filter(b => b.bookingId.includes(year)).length + 1).toString().padStart(3, '0');

			bookingData.push({
				id: newId,
				bookingId: `BKG-${year}-${count}`,
				amenityId,
				amenityName: amenity.name,
				residentId: parseInt(residentId),
				residentName: residentMap[residentId].name,
				flat: residentMap[residentId].flat,
				bookingDate,
				startTime,
				endTime,
				duration,
				guests,
				requests,
				status: 'pending',
				amount,
				paymentStatus: 'pending'
			});

			// Update amenity total bookings
			const amenityIndex = amenityData.findIndex(a => a.id === amenityId);
			if (amenityIndex !== -1) {
				amenityData[amenityIndex].totalBookings = (amenityData[amenityIndex].totalBookings || 0) + 1;
			}

			closeModal('bookingModal');
			loadTodaysBookings();
			initAmenitiesChart();
			updateStats();
			showNotification('Booking confirmed successfully!', 'success');
		}

		function saveAmenity() {
			const name = document.getElementById('name').value;
			const description = document.getElementById('description').value;
			const category = document.getElementById('category').value;
			const icon = document.getElementById('icon').value;
			const price = document.getElementById('price').value;
			const status = document.getElementById('status').value;

			if (!name || !description || !category || !icon || price === '' || !status) {
				showNotification('Please fill all required fields', 'error');
				return;
			}

			const obj = {
				name,
				description,
				category,
				icon,
				capacity: document.getElementById('capacity').value ? parseInt(document.getElementById('capacity').value) : null,
				price: parseInt(price),
				location: document.getElementById('location').value,
				status,
				openingTime: document.getElementById('openingTime').value,
				closingTime: document.getElementById('closingTime').value,
				rules: document.getElementById('rules').value,
				totalBookings: 0,
				rating: 0
			};

			const id = document.getElementById('amenityId').value;

			if (id) {
				const idx = amenityData.findIndex(a => a.id == id);
				if (idx !== -1) {
					amenityData[idx] = { ...amenityData[idx], ...obj, id: parseInt(id), amenityId: amenityData[idx].amenityId };
					showNotification('Amenity updated!', 'success');
				}
			} else {
				const newId = amenityData.length > 0 ? Math.max(...amenityData.map(a => a.id), 0) + 1 : 1;
				const year = new Date().getFullYear();
				const count = (amenityData.filter(a => a.amenityId.includes(year)).length + 1).toString().padStart(3, '0');
				amenityData.push({
					id: newId,
					amenityId: `AMN-${year}-${count}`,
					...obj
				});
				showNotification('Amenity added!', 'success');
			}

			closeModal('amenityFormModal');
			loadAmenitiesGrid();
			loadAmenitiesTable();
			filterAmenities();
			updateStats();
		}

		function exportAmenities() {
			let csv = 'Amenity ID,Name,Category,Capacity,Price,Status,Location,Total Bookings\n';
			amenityData.forEach(a => {
				csv += `${a.amenityId},${a.name},${a.category},${a.capacity || ''},${a.price},${a.status},${a.location || ''},${a.totalBookings || 0}\n`;
			});

			const blob = new Blob([csv], { type: 'text/csv' });
			const url = window.URL.createObjectURL(blob);
			const a = document.createElement('a');
			a.href = url;
			a.download = `amenities_${new Date().toISOString().split('T')[0]}.csv`;
			a.click();
			window.URL.revokeObjectURL(url);
			showNotification('Exported successfully!', 'success');
		}

		function showNotification(m, t = 'success') {
			const n = document.createElement('div');
			n.className = `notification ${t}`;
			n.innerHTML = `<i class="fas ${t === 'success' ? 'fa-check-circle' : t === 'error' ? 'fa-exclamation-circle' : 'fa-info-circle'}"></i><span>${m}</span>`;
			document.body.appendChild(n);
			setTimeout(() => { n.style.animation = 'slideOut 0.3s ease'; setTimeout(() => n.remove(), 300); }, 3000);
		}
